jQuery Geocomplete

A simple plugin for completing address forms.
View Github


Appended To Parent 
              $("#txtAddress1").geocomplete({
                appendToParent: true,
                fields: {
                  ...
                }
              });
Not Appended To Parent 
              $("#txtAddress2").geocomplete({
                appendToParent: false,
                fields: {
                  ...
                }
              });
Short Values 
              $("#txtAddress3").geocomplete({
                fields: {
                  txtAddress3 : "short street address",
                  txtCity3    : "short city",
                  txtCounty3  : "short county",
                  txtState3   : "short state",
                  ddlCountry3 : "short country",
                  txtZipCode3 : "short zipcode"
                }
              });
Strict Bounds 
              $("#txtAddress4").geocomplete({
                geolocate    : true,
                strictBounds : true,
                fields: {
                  ...
                }
              });
